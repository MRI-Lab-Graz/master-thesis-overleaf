% apa_template_settings.tex
% Maintainer preamble for thesis wrappers (shared settings)
% \documentclass[man]{apa7}

% Basic packages
\usepackage{float}
\usepackage[utf8]{inputenc}
\usepackage{setspace}
\usepackage{graphicx}
\usepackage{csquotes}
\usepackage{ifthen}

% --- Language handling ---
% If the wrapper defines \thesislanguage it will be used; default ngerman.
\providecommand{\thesislanguage}{ngerman}
\ifthenelse{\equal{\thesislanguage}{english}}{%
  \usepackage[english]{babel}
}{%
  \usepackage[ngerman]{babel}
}

% biblatex (APA) and language mapping
\usepackage[style=apa,sortcites=true,sorting=nyt,backend=biber]{biblatex}
\ifthenelse{\equal{\thesislanguage}{english}}{%
  \DeclareLanguageMapping{american}{american-apa}
}{%
  \DeclareLanguageMapping{german}{german-apa}
}
\addbibresource{bibliography.bib}

% Micro-typography and PDF metadata
\usepackage{microtype}
\usepackage{bookmark}
% apa7 already loads hyperref; set options via \hypersetup instead of reloading
\hypersetup{
  pdftitle={\thesistitle},
  pdfauthor={\thesisauthor},
  pdfsubject={\thesissubtitle},
  pdfkeywords={},
  pdfcreator={LaTeX with apa7},
  pdfproducer={pdfTeX}
}

% Header settings
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\makeatletter
\@ifundefined{shorttitle}{%
  \fancyhead[L]{\thesistitle}%
  \@ifundefined{@shorttitle}{\gdef\@shorttitle{\thesistitle}}{}
}{%
  \fancyhead[L]{\shorttitle}%
}
\@ifundefined{@shortauthor}{\gdef\@shortauthor{\thesisauthor}}{}
\fancyhead[R]{\thepage}
\renewcommand{\headrulewidth}{0pt}
\makeatother

% Document-wide formatting
\doublespacing
\usepackage{lipsum}

\newcommand{\thesistype}{Masterarbeit}
\newcommand{\thesisaffiliation}{zur Erlangung des akademischen Grades Master of Science\\
an der Naturwissenschaftlichen Fakultaet\\
der Karl-Franzens-Universitaet Graz}
\newcommand{\thesislogo}{figures/uni-graz-logo.pdf}

% === Title Page Macro ===
\newcommand{\maketitlepage}{%
  \begin{titlepage}
    \centering
    \vspace*{1cm}
    \large
    	extbf{\ifdefined\thesisauthor\thesisauthor\else Author Name\fi}

    \vspace{1.5cm}
    \Large
    	extbf{\ifdefined\thesistitle\thesistitle\else Title of the thesis\fi}

    \vspace{0.5cm}
    \normalsize
    	extbf{\thesissubtitle}

    \vspace{1.5cm}
    \Large
    	extbf{\thesistype}

    \vspace{0.5cm}
    \normalsize
    	hesisaffiliation

    \vspace{2cm}
    \includegraphics[width=0.2\textwidth]{\thesislogo}

    \vspace{1.5cm}
    Begutachter: \thesisadvisor\\
    	hesisinstitute\\
    	hesisdate
  \end{titlepage}
}
