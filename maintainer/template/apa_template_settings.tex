% Contains only preamble settings for the thesis
% \documentclass[man]{apa7}
% Basic packages
\usepackage{float}
\usepackage[utf8]{inputenc}
\usepackage{setspace}
\usepackage{graphicx}
\usepackage{csquotes}
\usepackage{ifthen}

% --- Language handling ---
% If the main file (thesis.tex) defines \thesislanguage it will be used.
% Default to ngerman. Note: do NOT use the primitive \language name.
\providecommand{\thesislanguage}{ngerman}
\ifthenelse{\equal{\thesislanguage}{english}}{
  \usepackage[english]{babel}
}{
  \usepackage[ngerman]{babel}
}

% biblatex (APA) and language mapping
\usepackage[style=apa,sortcites=true,sorting=nyt,backend=biber]{biblatex}
\ifthenelse{\equal{\thesislanguage}{english}}{
  \DeclareLanguageMapping{american}{american-apa}
}{
  \DeclareLanguageMapping{german}{german-apa}
}
\addbibresource{bibliography.bib}

% Improve micro-typography and PDF metadata for beginners
\usepackage{microtype} % better character protrusion and font expansion
\usepackage{bookmark}  % better PDF bookmarks
% apa7 already loads hyperref; set options via \hypersetup instead of reloading
\hypersetup{
  pdftitle={\thesistitle},
  pdfauthor={\thesisauthor},
  pdfsubject={\thesissubtitle},
  pdfkeywords={},
  pdfcreator={LaTeX with apa7},
  pdfproducer={pdfTeX}
}

% Header settings
\pagestyle{fancy}
\fancyhf{} % Clear header and footer
% Do NOT override class commands (e.g. \shorttitle) which may expect
% different argument semantics. Instead use them if they exist, otherwise
% fall back to the full title/author.
\makeatletter
% If \shorttitle (usually set by \shorttitle{...}) is undefined, use
% the full title; otherwise use the existing \shorttitle.
\@ifundefined{shorttitle}{%
  \fancyhead[L]{\thesistitle}%
  % Provide the internal \@shorttitle used by some classes/packages
  \@ifundefined{@shorttitle}{\gdef\@shorttitle{\thesistitle}}{}
}{%
  \fancyhead[L]{\shorttitle}%
}
% Also ensure \@shortauthor exists (some title implementations use it)
\@ifundefined{@shortauthor}{\gdef\@shortauthor{\thesisauthor}}{}
% Keep page number on the right
\fancyhead[R]{\thepage}
\renewcommand{\headrulewidth}{0pt} % Remove horizontal bar
\makeatother

% Document-wide formatting
\doublespacing
\usepackage{lipsum}
